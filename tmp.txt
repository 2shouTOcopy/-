#include <cmath>
#include <cctype>
#include <iomanip>
#include <limits>
#include <sstream>
#include <string>

class FloatFormatter : public CFormatter
{
public:
    FloatFormatter(float v, const std::string& fmt, const std::string& sep)
        : CFormatter(fmt, sep), value_(static_cast<double>(v))
    {
    }

    std::string format() override
    {
        FmtSpec spec;
        if (!ParseFmtSpec(formatStr, spec))
        {
            // Fallback: keep safe defaults if fmt is malformed
            spec.width = 1;
            spec.precision = 3;
        }

        if (std::isnan(value_)) return "nan";
        if (std::isinf(value_)) return (value_ < 0) ? "-inf" : "inf";

        const bool isNegative = std::signbit(value_);
        const double absValue = std::fabs(value_);

        const long long scale = Pow10(spec.precision);

        // Core fix:
        // Round after scaling so carry is correctly applied.
        // Example: 91.999913 with precision=3:
        // absValue*1000 = 91999.913 -> llround = 92000 -> 92.000
        const long long scaled = llround(absValue * static_cast<double>(scale));

        const long long integerPart = (spec.precision == 0) ? scaled : (scaled / scale);
        const long long fracPart    = (spec.precision == 0) ? 0     : (scaled % scale);

        std::string intStr = FormatInteger(integerPart, spec.width);

        // If rounding leads to exact 0, avoid "-0.000"
        if (isNegative && (integerPart != 0 || fracPart != 0))
        {
            intStr = "-" + intStr;
        }

        if (spec.precision == 0)
        {
            return intStr;
        }

        const std::string fracStr = FormatFraction(fracPart, spec.precision);
        return intStr + "." + fracStr;
    }

private:
    struct FmtSpec
    {
        int width = 1;
        int precision = 3;
    };

    // Parse format like: "%1.3f", "%05.2f", "%8.0f"
    static bool ParseFmtSpec(const std::string& fmt, FmtSpec& out)
    {
        // Remove spaces
        std::string s;
        s.reserve(fmt.size());
        for (char c : fmt)
        {
            if (!std::isspace(static_cast<unsigned char>(c)))
                s.push_back(c);
        }

        if (s.size() < 4) return false;
        if (s.front() != '%') return false;
        if (s.back() != 'f' && s.back() != 'F') return false;

        // Strip '%' and trailing 'f'
        s.erase(s.begin());
        s.pop_back();

        // Parse width (digits until '.')
        size_t i = 0;
        int width = 0;
        bool hasWidth = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasWidth = true;
            width = width * 10 + (s[i] - '0');
            ++i;
        }
        if (!hasWidth) return false;
        if (i >= s.size() || s[i] != '.') return false;
        ++i; // skip '.'

        // Parse precision (digits)
        int precision = 0;
        bool hasPrecision = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasPrecision = true;
            precision = precision * 10 + (s[i] - '0');
            ++i;
        }
        if (!hasPrecision) return false;
        if (i != s.size()) return false;

        if (width < 0) width = 0;
        if (precision < 0) precision = 0;
        if (precision > 18) precision = 18; // keep scale in long long safe range

        out.width = width;
        out.precision = precision;
        return true;
    }

    static long long Pow10(int n)
    {
        long long r = 1;
        for (int i = 0; i < n; ++i) r *= 10;
        return r;
    }

    static std::string FormatInteger(long long v, int width)
    {
        std::ostringstream oss;
        oss << std::setw(width) << std::setfill('0') << v; // keep your original behavior
        return oss.str();
    }

    static std::string FormatFraction(long long frac, int precision)
    {
        std::ostringstream oss;
        oss << std::setw(precision) << std::setfill('0') << frac;
        return oss.str();
    }

private:
    double value_;
};
