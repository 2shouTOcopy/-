class CFormatter {
protected:
  std::string formatStr; // 用于存储格式字符串
  std::string separator; // 用于存储分隔符

public:
  CFormatter(const std::string &fmt, const std::string &sep)
      : formatStr(fmt), separator(sep) {}

  virtual ~CFormatter() {}

  virtual std::string format() = 0;

  const std::string &getSeparator() const { return separator; }
};

// int 类型的 CFormatter
class IntFormatter : public CFormatter {
  int value;

public:
  IntFormatter(int v, const std::string &fmt, const std::string &sep)
      : CFormatter(fmt, sep), value(v) {}

  std::string format() override {
    int width = 0;
    sscanf(formatStr.c_str(), "%%%dd", &width);
    std::ostringstream oss;
    oss << std::setw(width) << std::setfill('0') << value;
    return oss.str();
  }
};
// float 类型的 CFormatter
class FloatFormatter : public CFormatter {
  float value;

public:
  FloatFormatter(float v, const std::string &fmt, const std::string &sep)
      : CFormatter(fmt, sep), value(v) {}

  std::string format() override {
    int width = 0, precision = 0;
    sscanf(formatStr.c_str(), "%%%d.%d", &width, &precision);

    bool isNegative = value < 0;
    float absValue = std::abs(value);

    long long integerPart = static_cast<long long>(absValue);

    float fractional = absValue - integerPart;

    std::ostringstream fracStream;
    fracStream << std::fixed << std::setprecision(precision) << fractional;
    std::string fracStr = fracStream.str();

    size_t dotPos = fracStr.find('.');
    if (dotPos != std::string::npos) {
      // Detect carry from rounding (e.g., 0.9996 with precision 3 -> "1.000")
      int fracIntPart = std::atoi(fracStr.substr(0, dotPos).c_str());
      if (fracIntPart >= 1) {
        integerPart += fracIntPart;
      }
      if (fracStr.length() > dotPos + 1) {
        fracStr = fracStr.substr(dotPos + 1);
      }
    }

    if (fracStr.length() > static_cast<size_t>(precision)) {
      fracStr = fracStr.substr(0, precision);
    } else {
      fracStr.append(precision - fracStr.length(), '0');
    }

    // 格式化整数部分 (左侧零填充)
    std::ostringstream intStream;
    intStream << std::setw(width) << std::setfill('0') << integerPart;
    std::string intStr = intStream.str();

    if (isNegative) {
      intStr = "-" + intStr;
    }

    return intStr + "." + fracStr;
  }
};

// 字符串类型的 CFormatter
class StringFormatter : public CFormatter {
  std::string value;

public:
  StringFormatter(const std::string &v, const std::string &fmt,
                  const std::string &sep)
      : CFormatter(fmt, sep), value(v) {}
  std::string format() override { return value; }
};

// 创建 CFormatter 的模板函数
template <typename T>
std::unique_ptr<CFormatter> createFormatter(T value, const std::string &fmt,
                                            const std::string &sep);

template <>
std::unique_ptr<CFormatter> createFormatter<int>(int value,
                                                 const std::string &fmt,
                                                 const std::string &sep) {
  return std::make_unique<IntFormatter>(value, fmt, sep);
}

template <>
std::unique_ptr<CFormatter> createFormatter<float>(float value,
                                                   const std::string &fmt,
                                                   const std::string &sep) {
  return std::make_unique<FloatFormatter>(value, fmt, sep);
}

template <>
std::unique_ptr<CFormatter>
createFormatter<std::string>(std::string value, const std::string &fmt,
                             const std::string &sep) {
  return std::make_unique<StringFormatter>(value, fmt, sep);
}
