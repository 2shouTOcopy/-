#include <cmath>
#include <cctype>
#include <iomanip>
#include <limits>
#include <sstream>
#include <string>

class CFormatter
{
protected:
    std::string formatStr;
    std::string sepStr;
public:
    CFormatter(const std::string& fmt, const std::string& sep) : formatStr(fmt), sepStr(sep) {}
    virtual ~CFormatter() = default;
    virtual std::string format() = 0;
};

class FloatFormatter : public CFormatter
{
    double value; // 用 double 做计算，避免 float 误差放大

    static bool parseWidthPrecision(const std::string& fmt, int& width, int& precision)
    {
        // 支持 "1.3" / "%1.3f" / "%01.3f" / "  %8.2f " 等
        std::string s;
        s.reserve(fmt.size());
        for (char c : fmt)
        {
            if (!std::isspace(static_cast<unsigned char>(c)))
                s.push_back(c);
        }

        // 去掉开头 '%'
        if (!s.empty() && s[0] == '%') s.erase(s.begin());

        // 去掉末尾类型字符，比如 'f'
        if (!s.empty() && (s.back() == 'f' || s.back() == 'F'))
            s.pop_back();

        // 允许前导 0（比如 01.3）
        // 手写解析：读取 width（可为空），遇到 '.' 再读 precision（可为空）
        width = 0;
        precision = 0;

        size_t i = 0;
        bool hasWidth = false;

        // width
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasWidth = true;
            width = width * 10 + (s[i] - '0');
            ++i;
        }

        if (i >= s.size() || s[i] != '.')
        {
            // 没有 '.' 就认为解析失败（你当前需求是 width.precision）
            return false;
        }
        ++i; // skip '.'

        bool hasPrec = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasPrec = true;
            precision = precision * 10 + (s[i] - '0');
            ++i;
        }

        return hasWidth && hasPrec;
    }

public:
    FloatFormatter(float v, const std::string& fmt, const std::string& sep)
        : CFormatter(fmt, sep), value(static_cast<double>(v)) {}

    std::string format() override
    {
        int width = 0, precision = 0;
        if (!parseWidthPrecision(formatStr, width, precision))
        {
            // 兜底：如果 formatStr 不是 "w.p" 格式，就给一个安全默认值
            width = 1;
            precision = 3;
        }

        // 特殊值处理
        if (std::isnan(value)) return "nan";
        if (std::isinf(value)) return (value < 0) ? "-inf" : "inf";

        bool isNegative = std::signbit(value);
        double absValue = std::fabs(value);

        // 防止 precision 太大导致 10^precision 溢出
        if (precision < 0) precision = 0;
        if (precision > 18) precision = 18; // long long 安全范围内的实用上限

        long long scale = 1;
        for (int i = 0; i < precision; ++i) scale *= 10;

        // 关键：这里做四舍五入，并自动处理 91.999 -> 92.000 的进位
        long long scaled = llround(absValue * static_cast<double>(scale));
        long long integerPart = (scale == 0) ? scaled : (scaled / scale);
        long long fracPart = (scale == 0) ? 0 : (scaled % scale);

        // 整数部分（左侧 0 填充）
        std::ostringstream intStream;
        intStream << std::setw(width) << std::setfill('0') << integerPart;
        std::string intStr = intStream.str();

        // 小数部分（固定 precision 位，左侧补 0）
        std::ostringstream fracStream;
        if (precision > 0)
        {
            fracStream << std::setw(precision) << std::setfill('0') << fracPart;
        }

        // 负号：注意 -0.xxx 也要带 '-'
        if (isNegative && (integerPart != 0 || fracPart != 0))
            intStr = "-" + intStr;

        if (precision == 0) return intStr;
        return intStr + "." + fracStream.str();
    }
};
