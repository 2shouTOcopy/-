#!/usr/bin/env python
# -*- coding: utf-8 -*-

import os
import sys
import struct

MAGIC = 0x484b4e46

SRC_FILE = "digicap.dav"
DST_FILE = "2.bin"

# FirmwareCryptHeader:
# uint32_t magic;
# uint32_t crc;
# uint8_t  ver;
# uint8_t  resv1[3];
# uint8_t  iv[16];
# uint32_t offset;
# uint32_t uoffset;
# uint32_t resv2;
#
# 总大小: 40 字节
HEADER_FMT_LE = "<IIB3s16sIII"
HEADER_FMT_BE = ">IIB3s16sIII"
HEADER_SIZE = struct.calcsize(HEADER_FMT_LE)


def parse_header(header_buf):
    """
    解析 FirmwareCryptHeader。
    默认先按小端解析，如果 magic 不匹配，再尝试大端。
    """

    if len(header_buf) != HEADER_SIZE:
        raise Exception("header size invalid, need %d bytes, got %d bytes" %
                        (HEADER_SIZE, len(header_buf)))

    fields = struct.unpack(HEADER_FMT_LE, header_buf)
    magic = fields[0]

    if magic == MAGIC:
        endian = "little"
        return endian, fields

    fields = struct.unpack(HEADER_FMT_BE, header_buf)
    magic = fields[0]

    if magic == MAGIC:
        endian = "big"
        return endian, fields

    raise Exception("invalid magic: little=0x%08x, big=0x%08x" %
                    (struct.unpack(HEADER_FMT_LE, header_buf)[0],
                     struct.unpack(HEADER_FMT_BE, header_buf)[0]))


def copy_from_offset(src_file, dst_file, start_offset):
    """
    从 src_file 的 start_offset 偏移处开始复制数据到 dst_file。
    """

    file_size = os.path.getsize(src_file)

    if start_offset > file_size:
        raise Exception("start_offset %d > file_size %d" %
                        (start_offset, file_size))

    with open(src_file, "rb") as fin:
        fin.seek(start_offset, os.SEEK_SET)

        with open(dst_file, "wb") as fout:
            while True:
                buf = fin.read(1024 * 1024)
                if not buf:
                    break
                fout.write(buf)


def main():
    if not os.path.exists(SRC_FILE):
        print("source file not found: %s" % SRC_FILE)
        return 1

    with open(SRC_FILE, "rb") as f:
        header_buf = f.read(HEADER_SIZE)

    endian, fields = parse_header(header_buf)

    magic = fields[0]
    crc = fields[1]
    ver = fields[2]
    iv = fields[4]
    offset = fields[5]
    uoffset = fields[6]
    resv2 = fields[7]

    start_offset = HEADER_SIZE + uoffset

    print("header size : %d" % HEADER_SIZE)
    print("endian      : %s" % endian)
    print("magic       : 0x%08x" % magic)
    print("crc         : 0x%08x" % crc)
    print("ver         : %d" % ver)
    print("offset      : %d" % offset)
    print("uoffset     : %d" % uoffset)
    print("resv2       : 0x%08x" % resv2)
    print("start offset: %d" % start_offset)

    copy_from_offset(SRC_FILE, DST_FILE, start_offset)

    print("generate %s success" % DST_FILE)
    return 0


if __name__ == "__main__":
    sys.exit(main())
