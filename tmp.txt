#include <iostream>
#include <sstream>
#include <vector>
#include <iomanip>
#include <string>
#include <map>
#include <cstdint>

// 定义等级映射表
std::map<std::string, uint8_t> level_map = {
    {"0", 0x00}, {"1+", 0x01}, {"2+", 0x02}, {"3+", 0x03}, {"4+", 0x04}, {"5+", 0x05}, {"FF", 0xFF}
};

// 将试管号转换为四字节
std::vector<uint8_t> convertTubeNumber(int tubeNumber) {
    std::vector<uint8_t> tubeBytes;
    for (int i = 3; i >= 0; --i) {
        tubeBytes.push_back((tubeNumber >> (i * 8)) & 0xFF);
    }
    return tubeBytes;
}

// 将单图像识别结果的字符串解析为十六进制数据
std::vector<uint8_t> parseRecognitionResult(const std::string &group) {
    std::vector<uint8_t> result;
    std::istringstream ss(group);
    std::string segment;

    // 以分号分割，获取每个部分
    while (std::getline(ss, segment, ';')) {
        if (segment == "null") {
            result.push_back(0xFF);
            result.push_back(0xFF);
        } else {
            std::istringstream segmentStream(segment);
            std::string name, level, scoreStr;

            if (std::getline(segmentStream, name, '@') &&
                std::getline(segmentStream, level, '@') &&
                std::getline(segmentStream, scoreStr)) {

                uint8_t levelHex = level_map.count(level) ? level_map[level] : 0xFF;
                result.push_back(levelHex);

                try {
                    float score = std::stof(scoreStr);
                    uint8_t scoreHex = static_cast<uint8_t>(score);
                    result.push_back(scoreHex);
                } catch (...) {
                    result.push_back(0xFF);
                }
            }
        }
    }
    return result;
}

// 计算校验码（不包含开头的 0x85）
uint8_t calculateChecksum(const std::vector<uint8_t> &data) {
    uint32_t sum = 0;
    // 跳过第一个字节（0x85），从 data[1] 开始计算
    for (size_t i = 1; i < data.size(); ++i) {
        sum += data[i];
    }
    return static_cast<uint8_t>(sum & 0xFF);  // 保留低8位
}

// 解析完整输入字符串
std::vector<uint8_t> parseInputString(const std::string &input) {
    std::vector<uint8_t> buf;
    std::istringstream ss(input);
    std::string segment;

    // 解析试管号
    std::getline(ss, segment, ';');
    int tubeNumber = std::stoi(segment);
    std::vector<uint8_t> tubeBytes = convertTubeNumber(tubeNumber);

    // 解析图像索引
    std::vector<int> imageIndices;
    for (int i = 0; i < 3; ++i) {
        std::getline(ss, segment, ';');
        imageIndices.push_back(std::stoi(segment));
    }

    // 解析每个图像索引的识别数据
    for (size_t i = 0; i < imageIndices.size(); ++i) {
        std::string recognitionGroup;
        std::vector<uint8_t> groupData;

        // 每组数据开头的标识符
        groupData.push_back(0x85);

        // 插入试管号数据和图像索引
        groupData.insert(groupData.end(), tubeBytes.begin(), tubeBytes.end());
        groupData.push_back(static_cast<uint8_t>(imageIndices[i]));

        // 将完整的三部分识别数据作为一个段读取
        for (int j = 0; j < 3; ++j) {
            if (j > 0) recognitionGroup += ";";
            std::getline(ss, segment, ';');
            recognitionGroup += segment;
        }

        // 解析并插入该图像的识别结果
        std::vector<uint8_t> recognitionData = parseRecognitionResult(recognitionGroup);
        groupData.insert(groupData.end(), recognitionData.begin(), recognitionData.end());

        // 计算并添加校验码
        uint8_t checksum = calculateChecksum(groupData);
        groupData.push_back(checksum);

        // 添加结束标识符
        groupData.push_back(0xAA);

        // 将完整的组数据加入到主缓冲区
        buf.insert(buf.end(), groupData.begin(), groupData.end());
    }
    return buf;
}

// 输出结果
void printHexBuffer(const std::vector<uint8_t> &buffer) {
    for (uint8_t byte : buffer) {
        std::cout << "0x" << std::hex << std::setw(2) << std::setfill('0') << (int)byte << " ";
    }
    std::cout << std::dec << std::endl;
}

int main() {
    std::string input = "555;0;6;9;黄疸@1+@94.410004;null;null;null;脂血@2+@94.410004;null;null;null;溶血@3+@94.410004";
    std::vector<uint8_t> hexBuffer = parseInputString(input);

    // 输出结果
    printHexBuffer(hexBuffer);

    return 0;
}
