#include <string>
#include <regex>
#include <cerrno>
#include <climits>
#include <cstdlib>

// 你已有：LOGI/LOGE、SubParam、SUB_PARAM_REGULAR_EXP

// 仅允许非负 int 的安全解析：不抛异常
static bool ParseNonNegIntSafe(const std::string& s, int& out)
{
    if (s.empty()) return false;

    // 只允许纯数字
    for (size_t i = 0; i < s.size(); ++i)
    {
        if (s[i] < '0' || s[i] > '9')
            return false;
    }

    // 防御：数字位数过长基本一定溢出/脏数据
    // INT_MAX=2147483647（10位），允许 10~11 位都可以，但过长没意义
    if (s.size() > 11) return false;

    errno = 0;
    char* end = nullptr;
    long long v = std::strtoll(s.c_str(), &end, 10);

    if (end == nullptr || *end != '\0') return false;
    if (errno == ERANGE) return false;
    if (v < 0 || v > INT_MAX) return false;

    out = static_cast<int>(v);
    return true;
}

// suffix 校验：允许 "*" 或 非负 int（不抛异常）
// 如果 suffix 是数字，会额外把它解析成 int 放到 outIndex（仅用于校验，调用者不需要保存也行）
static bool ValidateSuffix(const std::string& suffix, bool& isAll, int& outIndex)
{
    isAll = false;
    outIndex = -1;

    if (suffix == "*")
    {
        isAll = true;
        return true;
    }

    int idx = -1;
    if (!ParseNonNegIntSafe(suffix, idx))
        return false;

    outIndex = idx;
    return true;
}

int CAlgoUtils::ParseSubscribeItem(const std::string &str, SubParam &item)
{
    if (str.empty())
        return -1;

    try
    {
        // 1) 使用 regex_match：要求整串符合格式（比 regex_search 更安全）
        static const std::regex re(SUB_PARAM_REGULAR_EXP);
        std::smatch match;

        // 你的正则有 5 个捕获组，所以 match.size() 必须是 6（含 group0）
        if (!std::regex_match(str, match, re) || match.size() != 6)
        {
            LOGE("no match: %s\n", str.c_str());
            return -2;
        }

        // 2) algoID 强校验：避免 stoi 抛异常导致进程 abort
        {
            int algoId = -1;
            const std::string sAlgo = match.str(1);
            if (!ParseNonNegIntSafe(sAlgo, algoId))
            {
                LOGE("invalid algoID: '%s', src='%s'\n", sAlgo.c_str(), str.c_str());
                return -4;
            }
            item.algoID = algoId;
        }

        item.moduleName  = match.str(2);
        item.paramName   = match.str(3);
        item.printFormat = match.str(4);
        item.suffix      = match.str(5); // 必有：数字 or "*"

        // 3) suffix 强校验：必须是 "*" 或 非负 int，并且在 int 范围内
        {
            bool isAll = false;
            int idx = -1;
            if (!ValidateSuffix(item.suffix, isAll, idx))
            {
                LOGE("invalid suffix: '%s', src='%s'\n", item.suffix.c_str(), str.c_str());
                return -5;
            }

            // 可选：如果你这里就知道 nCount（比如配置阶段已知），可以在此处做范围检查：
            // if (!isAll && idx >= nCount) return -6;
            // 但如果 nCount 只有运行时才知道，就不要在这里做。
        }

        LOGI("algoID: %d\n", item.algoID);
        LOGI("moduleName: %s\n", item.moduleName.c_str());
        LOGI("paramName: %s\n", item.paramName.c_str());
        LOGI("printFormat: %s\n", item.printFormat.c_str());
        LOGI("suffix: %s\n", item.suffix.c_str());

        return 0;
    }
    catch (const std::regex_error& e)
    {
        LOGE("regex_error caught: %s\n", e.what());
        return -3;
    }
}
