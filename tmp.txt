#include <cmath>
#include <cctype>
#include <cstdio>
#include <limits>
#include <string>

class FloatFormatter : public CFormatter
{
public:
    FloatFormatter(float v, const std::string& fmt, const std::string& sep)
        : CFormatter(fmt, sep), value_(v) {}

    std::string format() override
    {
        FmtSpec spec;
        if (!ParseFmtSpec(formatStr, spec))
        {
            spec.width = 1;
            spec.precision = 3;
        }

        if (std::isnan(value_)) return "nan";
        if (std::isinf(value_)) return (value_ < 0) ? "-inf" : "inf";

        const bool isNeg = std::signbit(value_);
        const double absV = std::fabs(static_cast<double>(value_));

        // 1) let snprintf handle rounding and formatting to fixed decimal digits
        const std::string rounded = ToFixed(absV, spec.precision); // e.g. "12.000" / "91.124"

        // 2) split integer/fraction
        std::string intPart, fracPart;
        SplitFixed(rounded, spec.precision, intPart, fracPart);

        // 3) decide if we should print '-' (avoid "-0.000")
        const bool isZero = IsAllZeros(intPart) && IsAllZeros(fracPart);
        const bool printNeg = isNeg && !isZero;

        // 4) integer padding:
        // width is for "integer area including sign" in your expected behavior.
        const int padWidth = printNeg ? (spec.width - 1) : spec.width;
        const std::string paddedInt = PadLeftZeros(intPart, padWidth);

        // 5) combine
        std::string out;
        out.reserve(32);

        if (printNeg) out.push_back('-');
        out += paddedInt;

        if (spec.precision > 0)
        {
            out.push_back('.');
            out += fracPart;
        }

        return out;
    }

private:
    struct FmtSpec
    {
        int width = 1;      // integer area width INCLUDING sign if negative
        int precision = 3;  // digits after '.'
    };

    // Parse "%<width>.<precision>f" (e.g. "%8.3f", "%1.0f", "%05.2f")
    static bool ParseFmtSpec(const std::string& fmt, FmtSpec& out)
    {
        std::string s;
        s.reserve(fmt.size());
        for (char c : fmt)
        {
            if (!std::isspace(static_cast<unsigned char>(c)))
                s.push_back(c);
        }

        if (s.size() < 4) return false;
        if (s.front() != '%') return false;
        if (s.back() != 'f' && s.back() != 'F') return false;

        s.erase(s.begin()); // remove '%'
        s.pop_back();       // remove 'f'

        size_t i = 0;

        int width = 0;
        bool hasWidth = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasWidth = true;
            width = width * 10 + (s[i] - '0');
            ++i;
        }
        if (!hasWidth) return false;
        if (i >= s.size() || s[i] != '.') return false;
        ++i;

        int precision = 0;
        bool hasPrec = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasPrec = true;
            precision = precision * 10 + (s[i] - '0');
            ++i;
        }
        if (!hasPrec) return false;
        if (i != s.size()) return false;

        if (width < 0) width = 0;
        if (precision < 0) precision = 0;
        if (precision > 18) precision = 18;

        out.width = width;
        out.precision = precision;
        return true;
    }

    static std::string ToFixed(double v, int precision)
    {
        char fmt[32];
        std::snprintf(fmt, sizeof(fmt), "%%.%df", precision);

        char buf[128];
        int n = std::snprintf(buf, sizeof(buf), fmt, v);
        if (n >= 0 && n < static_cast<int>(sizeof(buf)))
            return std::string(buf);

        if (n < 0) return "0";

        std::string out;
        out.resize(static_cast<size_t>(n) + 1);
        std::snprintf(out.data(), out.size(), fmt, v);
        out.pop_back(); // remove '\0'
        return out;
    }

    static void SplitFixed(const std::string& s, int precision,
                           std::string& intPart, std::string& fracPart)
    {
        const size_t dot = s.find('.');
        if (dot == std::string::npos)
        {
            intPart = s;
            fracPart.clear();
            if (precision > 0) fracPart.assign(static_cast<size_t>(precision), '0');
            return;
        }

        intPart = s.substr(0, dot);
        fracPart = (dot + 1 < s.size()) ? s.substr(dot + 1) : "";

        // keep exactly `precision` digits as a safety net
        if (precision <= 0)
        {
            fracPart.clear();
            return;
        }
        if (fracPart.size() < static_cast<size_t>(precision))
            fracPart.append(static_cast<size_t>(precision) - fracPart.size(), '0');
        else if (fracPart.size() > static_cast<size_t>(precision))
            fracPart.resize(static_cast<size_t>(precision));
    }

    static std::string PadLeftZeros(const std::string& s, int width)
    {
        if (width <= 0) return s;
        if (static_cast<int>(s.size()) >= width) return s;
        return std::string(static_cast<size_t>(width - static_cast<int>(s.size())), '0') + s;
    }

    static bool IsAllZeros(const std::string& s)
    {
        if (s.empty()) return true;
        for (char c : s)
        {
            if (c != '0') return false;
        }
        return true;
    }

private:
    float value_;
};
