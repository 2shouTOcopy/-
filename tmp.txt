int CAlgo::encMaskImg(const char *name,
                      unsigned int srcWidth, unsigned int srcHeight, void *srcData,
                      unsigned int dstWidth, unsigned int dstHeight,
                      Infra::DataBufferPtr &pImg)
{
    if (!name || !srcData || srcWidth == 0 || srcHeight == 0 || dstWidth == 0 || dstHeight == 0)
    {
        LOGE("invalid args: name=%p srcData=%p src=%ux%u dst=%ux%u\n",
             name, srcData, srcWidth, srcHeight, dstWidth, dstHeight);
        return IMVS_EC_ALGO_IMAGE_VENC;
    }

    IspImgProImgInfoEx image_info = {0};
    image_info.pixformat = ISP_IMG_FMT_PIXEL_FORMAT_YUV_400;
    image_info.width = srcWidth;
    image_info.height = srcHeight;
    image_info.step[0] = srcWidth;
    image_info.data[0] = srcData;
    image_info.phy[0] = MMZmemGetPhyAddr(image_info.data[0]);

    // 如果 phy 必须有效，可以直接强校验
    if (image_info.phy[0] == 0)
    {
        LOGW("srcData has no valid phy addr, srcData=%p, maybe not MMZ\n", srcData);
        // 如果你们 ISP 必须要 phy，这里建议 return error
        // return IMVS_EC_ALGO_IMAGE_VENC;
    }

    // 关键：把缩放后的 buffer 放在 if 外，保证生命周期覆盖到编码结束
    Infra::DataBufferPtr scaledImgBuf;

    // 源图像尺寸大于目标图像尺寸时，需要先进行降采样处理
    if (srcWidth > dstWidth || srcHeight > dstHeight)
    {
        IspImgProImgInfo stImageInfo = {ISP_IMG_FMT_JPEG, 0};
        stImageInfo.data[0] = srcData;
        stImageInfo.width = srcWidth;
        stImageInfo.height = srcHeight;
        stImageInfo.step[0] = srcWidth;
        stImageInfo.pixformat = static_cast<IspImgFormat>(ISP_IMG_FMT_PIXEL_FORMAT_YUV_400);

        // 如果你的 scaler 有 stride 对齐要求，这里应改成 stride*dstHeight
        unsigned int tmpImgLen = dstWidth * dstHeight;

        scaledImgBuf = Infra::DataBufferPtr(new Adapter::MMZPoolBuffer(tmpImgLen, 8, name));
        if (!scaledImgBuf || !scaledImgBuf->isValid())
        {
            LOGE("scaled buf invalid. size:%u align:%d name:%s\n", tmpImgLen, 8, name);
            return IMVS_EC_ALGO_IMAGE_VENC;
        }

        IspImgFillImgInfo tmpfillInfo = {nullptr, 0};
        tmpfillInfo.fill_width = dstWidth;
        tmpfillInfo.fill_height = dstHeight;
        tmpfillInfo.img_buf = scaledImgBuf->modifyBuffer();
        tmpfillInfo.src_img_len = scaledImgBuf->capacity();

        int ret = Parameter::IParamCtrl::getComponent()->getIspCtrl(0)->set(
            ISP_PT_IMG_SCALE_REENTRANT, &stImageInfo, sizeof(stImageInfo), &tmpfillInfo);
        if (ret < 0)
        {
            LOGE("raw scale fail ret:0x%x\n", ret);
            return IMVS_EC_ALGO_IMAGE_SCALE;
        }

        image_info.pixformat = ISP_IMG_FMT_PIXEL_FORMAT_YUV_400;
        image_info.width = dstWidth;
        image_info.height = dstHeight;
        image_info.step[0] = dstWidth;  // 若 scaler 输出 stride 对齐，这里也要同步
        image_info.data[0] = scaledImgBuf->modifyBuffer();
        image_info.phy[0] = MMZmemGetPhyAddr(image_info.data[0]);

        if (image_info.phy[0] == 0)
        {
            LOGE("scaled buffer has no valid phy addr, buf=%p\n", image_info.data[0]);
            return IMVS_EC_ALGO_IMAGE_VENC;
        }
    }

    // 输出 buffer：至少给 dstWidth*dstHeight 的下限，再加点冗余更安全（按你们 ISP 习惯调）
    unsigned int minRaw = dstWidth * dstHeight;
    unsigned int dstImgLen = minRaw + 64 * 1024; // 简单粗暴的安全余量（可按经验调整）

    Infra::DataBufferPtr pImgBuf = Infra::DataBufferPtr(new Adapter::MMZPoolBuffer(dstImgLen, 8, name));
    if (!pImgBuf || !pImgBuf->isValid())
    {
        LOGW("buf invalid. size:%u align:%d name:%s\n", dstImgLen, 8, name);
        return IMVS_EC_ALGO_IMAGE_VENC;
    }

    IspImgFillImgInfo fillInfo = {nullptr, 0};
    fillInfo.img_buf = pImgBuf->modifyBuffer();
    fillInfo.src_img_len = pImgBuf->capacity();
    fillInfo.fill_width = dstWidth;
    fillInfo.fill_height = dstHeight;
    fillInfo.fill_img_len = 0;

    int ret = Parameter::IParamCtrl::getComponent()->getIspCtrl(0)->set(
        ISP_PT_IMG_ENC_JPG_EX, &image_info, sizeof(image_info), &fillInfo);

    if (ret != 0)
    {
        LOGE("venc_jpeg fail ret:0x%x\n", ret);
        return IMVS_EC_ALGO_IMAGE_VENC;
    }

    if (fillInfo.fill_img_len == 0 || fillInfo.fill_img_len > pImgBuf->capacity())
    {
        LOGE("invalid jpeg len:%u capacity:%u\n", fillInfo.fill_img_len, pImgBuf->capacity());
        return IMVS_EC_ALGO_IMAGE_VENC;
    }

    pImgBuf->resize(fillInfo.fill_img_len);
    pImg = pImgBuf;
    return 0;
}
