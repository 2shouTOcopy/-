#include <stddef.h>
#include <string.h>
#include <stdio.h>

static int safe_strcpy(char *dst, size_t dstsz, const char *src)
{
    size_t n;
    if (!dst || dstsz == 0 || !src) return -1;

    n = strnlen(src, dstsz - 1);
    memcpy(dst, src, n);
    dst[n] = '\0';
    return (src[n] == '\0') ? 0 : -1; /* 截断则返回 -1 */
}

static int safe_copy_range(char *dst, size_t dstsz, const char *begin, const char *end)
{
    size_t len;
    if (!dst || dstsz == 0 || !begin || !end || end < begin) return -1;

    len = (size_t)(end - begin);
    if (len >= dstsz) len = dstsz - 1;

    memcpy(dst, begin, len);
    dst[len] = '\0';
    return ((size_t)(end - begin) < dstsz) ? 0 : -1;
}

static int split_whole_name_s(const char *whole_name,
                             char *fname, size_t fnamesz,
                             char *ext,   size_t extsz)
{
    const char *p_ext;

    if (!whole_name || !fname || !ext || fnamesz == 0 || extsz == 0)
        return -1;

    p_ext = strrchr(whole_name, '.');
    if (p_ext != NULL)
    {
        if (safe_strcpy(ext, extsz, p_ext) != 0) return -1;                 /* ".jpg" */
        if (safe_copy_range(fname, fnamesz, whole_name, p_ext) != 0) return -1; /* "abc" */
    }
    else
    {
        ext[0] = '\0';
        if (safe_strcpy(fname, fnamesz, whole_name) != 0) return -1;
    }
    return 0;
}

int splitpath_s(const char *path,
                char *drive, size_t drivesz,
                char *dir,   size_t dirsz,
                char *fname, size_t fnamesz,
                char *ext,   size_t extsz)
{
    const char *p_slash;
    size_t len;

    if (!drive || drivesz == 0 || !dir || dirsz == 0 ||
        !fname || fnamesz == 0 || !ext || extsz == 0)
        return -1;

    drive[0] = '\0';

    if (path == NULL)
    {
        dir[0] = '\0'; fname[0] = '\0'; ext[0] = '\0';
        return 0;
    }

    len = strlen(path);
    if (len == 0)
    {
        dir[0] = '\0'; fname[0] = '\0'; ext[0] = '\0';
        return 0;
    }

    /* 修复原 bug：最后一个字符是否 '/' */
    if (path[len - 1] == '/')
    {
        if (safe_strcpy(dir, dirsz, path) != 0) return -1;
        fname[0] = '\0';
        ext[0] = '\0';
        return 0;
    }

    p_slash = strrchr(path, '/');
    if (p_slash != NULL)
    {
        const char *whole = p_slash + 1;

        if (split_whole_name_s(whole, fname, fnamesz, ext, extsz) != 0) return -1;

        /* dir: 拷贝到 whole 之前（包含 '/'） */
        if (safe_copy_range(dir, dirsz, path, whole) != 0) return -1;
    }
    else
    {
        if (split_whole_name_s(path, fname, fnamesz, ext, extsz) != 0) return -1;
        dir[0] = '\0';
    }

    return 0;
}
