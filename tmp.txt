#include <thread>
#include <mutex>
#include <condition_variable>
#include <atomic>
#include <string>
#include <cstring>
#include <algorithm>

struct AsyncRecvExec
{
    struct Result {
        uint64_t seq = 0;
        int ret = -1;
        std::string text;
        bool valid = false;
    };

    std::thread th;
    std::mutex mtx;
    std::condition_variable cv;
    bool stop = false;

    // 单槽任务（覆盖式）
    bool hasTask = false;
    uint64_t taskSeq = 0;
    int taskModuleId = 0;
    std::string taskCmd;

    // 单槽结果
    Result result;

    std::atomic<uint64_t> seqGen{0};
    std::atomic<uint64_t> cancelFloor{0};

    AsyncRecvExec()
    {
        th = std::thread([this]{ loop(); });
    }
    ~AsyncRecvExec()
    {
        {
            std::lock_guard<std::mutex> lk(mtx);
            stop = true;
        }
        cv.notify_all();
        if (th.joinable()) th.join();
    }

    uint64_t Submit(int moduleId, const std::string& cmd)
    {
        uint64_t seq = ++seqGen;
        {
            std::lock_guard<std::mutex> lk(mtx);
            hasTask = true;
            taskSeq = seq;
            taskModuleId = moduleId;
            taskCmd = cmd;         // 覆盖旧任务：符合 PLC “只关心最新一次”
        }
        cv.notify_one();
        return seq;
    }

    void CancelAll()
    {
        cancelFloor.store(seqGen.load(std::memory_order_acquire), std::memory_order_release);
    }

    bool TryPop(Result& out)
    {
        std::lock_guard<std::mutex> lk(mtx);
        if (!result.valid) return false;
        out = result;
        result.valid = false;
        return true;
    }

    void loop()
    {
        for (;;)
        {
            uint64_t seq = 0;
            int moduleId = 0;
            std::string cmd;

            {
                std::unique_lock<std::mutex> lk(mtx);
                cv.wait(lk, [&]{ return stop || hasTask; });
                if (stop) return;

                seq = taskSeq;
                moduleId = taskModuleId;
                cmd = taskCmd;
                hasTask = false;
            }

            if (seq <= cancelFloor.load(std::memory_order_acquire))
                continue;

            std::string resultText;
            CCommProxy::MessageInfo mi{0};
            mi.moduleId = moduleId;

            mi.len = (int)std::min(cmd.size(), sizeof(mi.msg) - 1);
            memcpy(mi.msg, cmd.data(), mi.len);
            mi.msg[mi.len] = '\0';

            int ret = CCommProxy::getInstance()->SyncRecv(mi, resultText);

            if (seq <= cancelFloor.load(std::memory_order_acquire))
                continue;

            {
                std::lock_guard<std::mutex> lk(mtx);
                result.seq = seq;
                result.ret = ret;
                result.text = std::move(resultText);
                result.valid = true;
            }
        }
    }
};

// 文件内静态：全局唯一后台线程
static AsyncRecvExec g_async_exec;
static uint64_t g_pending_seq = 0;
static int g_pending = 0;
