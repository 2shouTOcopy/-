#include <cctype>
#include <cmath>
#include <cstdio>
#include <iomanip>
#include <limits>
#include <sstream>
#include <string>

class FloatFormatter : public CFormatter
{
public:
    FloatFormatter(float v, const std::string& fmt, const std::string& sep)
        : CFormatter(fmt, sep), value_(static_cast<double>(v))
    {
    }

    std::string format() override
    {
        FmtSpec spec;
        if (!ParseFmtSpec(formatStr, spec))
        {
            spec.width = 1;
            spec.precision = 3;
        }

        if (std::isnan(value_)) return "nan";
        if (std::isinf(value_)) return (value_ < 0) ? "-inf" : "inf";

        const bool isNeg = std::signbit(value_);
        const double absV = std::fabs(value_);

        // 1) Let libc do decimal rounding (matches printf semantics much better)
        //    Format: "%.3f" -> "91.124", "92.000", etc.
        std::string rounded = FormatBySnprintf(absV, spec.precision);

        // 2) Split integer/fractional part from the rounded decimal string
        std::string intPartStr;
        std::string fracPartStr;
        SplitDecimal(rounded, spec.precision, intPartStr, fracPartStr);

        // 3) Apply your custom integer left-zero-padding by width
        intPartStr = PadLeftZeros(intPartStr, spec.width);

        // 4) Apply sign: avoid "-0.000" when rounded result is exactly zero
        if (isNeg && !IsAllZero(intPartStr) /* after padding */)
        {
            intPartStr = "-" + intPartStr;
        }
        else if (isNeg && spec.precision > 0 && IsAllZero(intPartStr) && !IsAllZero(fracPartStr))
        {
            // e.g. -0.001 case: integer part is "0" but frac is non-zero
            intPartStr = "-" + intPartStr;
        }

        if (spec.precision == 0)
            return intPartStr;

        return intPartStr + "." + fracPartStr;
    }

private:
    struct FmtSpec
    {
        int width = 1;
        int precision = 3;
    };

    // Parse "%<width>.<precision>f" (e.g. "%1.3f", "%05.2f", "%8.0f")
    static bool ParseFmtSpec(const std::string& fmt, FmtSpec& out)
    {
        std::string s;
        s.reserve(fmt.size());
        for (char c : fmt)
        {
            if (!std::isspace(static_cast<unsigned char>(c)))
                s.push_back(c);
        }

        if (s.size() < 4) return false;
        if (s.front() != '%') return false;
        if (s.back() != 'f' && s.back() != 'F') return false;

        s.erase(s.begin()); // remove '%'
        s.pop_back();       // remove 'f'

        size_t i = 0;

        int width = 0;
        bool hasWidth = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasWidth = true;
            width = width * 10 + (s[i] - '0');
            ++i;
        }
        if (!hasWidth) return false;
        if (i >= s.size() || s[i] != '.') return false;
        ++i;

        int precision = 0;
        bool hasPrec = false;
        while (i < s.size() && std::isdigit(static_cast<unsigned char>(s[i])))
        {
            hasPrec = true;
            precision = precision * 10 + (s[i] - '0');
            ++i;
        }
        if (!hasPrec) return false;
        if (i != s.size()) return false;

        if (width < 0) width = 0;
        if (precision < 0) precision = 0;
        if (precision > 18) precision = 18;

        out.width = width;
        out.precision = precision;
        return true;
    }

    static std::string FormatBySnprintf(double v, int precision)
    {
        // Build format string like "%.3f"
        char fmt[32];
        std::snprintf(fmt, sizeof(fmt), "%%.%df", precision);

        // First try with a reasonable buffer; if not enough, resize.
        char buf[128];
        int n = std::snprintf(buf, sizeof(buf), fmt, v);
        if (n >= 0 && n < static_cast<int>(sizeof(buf)))
            return std::string(buf);

        // Need bigger buffer
        if (n < 0) return "0"; // very defensive fallback
        std::string out;
        out.resize(static_cast<size_t>(n) + 1);
        std::snprintf(out.data(), out.size(), fmt, v);
        out.pop_back(); // remove trailing '\0'
        return out;
    }

    static void SplitDecimal(const std::string& s,
                             int precision,
                             std::string& intPart,
                             std::string& fracPart)
    {
        const size_t dot = s.find('.');
        if (dot == std::string::npos)
        {
            // precision==0 commonly lands here
            intPart = s;
            fracPart.clear();
            return;
        }

        intPart = s.substr(0, dot);
        fracPart = (dot + 1 < s.size()) ? s.substr(dot + 1) : std::string();

        // Ensure exactly `precision` digits (printf should already do it, but keep safe)
        if (precision <= 0)
        {
            fracPart.clear();
            return;
        }
        if (fracPart.size() < static_cast<size_t>(precision))
            fracPart.append(static_cast<size_t>(precision) - fracPart.size(), '0');
        else if (fracPart.size() > static_cast<size_t>(precision))
            fracPart.resize(static_cast<size_t>(precision));
    }

    static std::string PadLeftZeros(const std::string& s, int width)
    {
        if (width <= 0) return s;
        if (static_cast<int>(s.size()) >= width) return s;
        return std::string(static_cast<size_t>(width - s.size()), '0') + s;
    }

    static bool IsAllZero(const std::string& s)
    {
        for (char c : s)
        {
            if (c != '0') return false;
        }
        return !s.empty();
    }

private:
    double value_;
};
